Queue using Array

Implementation

#include<iostream>
#include<cstdlib>
using namespace std;

struct Queue
{
    int size;
    int f;
    int r;
    int * arr;
};

int main(){
    struct Queue * q = (struct Queue *)malloc(sizeof(struct Queue));
    q->size = 10;
    q->f = -1;
    q->r = -1;
    q->arr = (int *)malloc(q->size*sizeof(int));
 return 0;
}

Functions


1. Traversal

void display(struct Queue * q){
    for (int i = q->f+1; i <= q->r; i++)
    {
        printf("%d ",q->arr[i]);
    }
    printf("\n");
}

2. isFull

int isFull(struct Queue * q){
    if (q->r == q->size-1)
    {
        return 1;
    }
    return 0;
}

3. isEmpty

int isEmpty(struct Queue * q){
    if (q->f == q->r)
    {
        return 1;
    }
    return 0;
}

4. Enqueue

void enqueue(struct Queue * q, int data){
    if (!isFull(q))
    {
        q->r++;
        q->arr[q->r] = data;
    }
    else{
        printf("Queue overflow!\n");
    }
}

5. Dequeue

int dequeue(struct Queue * q){
    int d = -1;
    if (!isEmpty(q))
    {
        q->f++;
        d = q->arr[q->f];
    }
    else{
        printf("Empty queue!\n");
    }
    return d;
}

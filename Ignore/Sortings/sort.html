1. Bubble sort

void bubbleSort(int * arr, int size){
    for (int i = 0; i < size-1; i++)
    {
        int a = 0;// to make bubble sort adaptive which it is not by defaut!
        for (int j = 0; j < size-1-i; j++)
        {
            if (arr[j]>arr[j+1])
            {
                printf("into 'j'\n");
                int temp = arr[j+1];
                arr[j+1] = arr[j];
                arr[j] = temp;
                a = 1;
            }
            
        }
        if (a == 0)
            {
                break;
            }
    }
    
}

2. Insertion sort

void insertionSort(int * arr, int size){
    int desired, j;
    for (int i = 1; i < size; i++)
    {
        desired = arr[i];
        j = i-1;
        while (j>=0 && desired<arr[j])
        {
            arr[j+1] = arr[j];
            j--;
        }
        arr[j+1] = desired;
    }
    
}

3. Selection sort

void selectionSort(int * arr, int size){
    int minInd, temp;
    for (int i = 0; i < size-1; i++)
    {
        minInd = i;
        for (int j = i+1; j < size; j++)
        {
            if (arr[j]<arr[minInd])
            {
                minInd = j;
            }
            
        }
        temp = arr[i];
        arr[i] = arr[minInd];
        arr[minInd] = temp;
    }
    
}

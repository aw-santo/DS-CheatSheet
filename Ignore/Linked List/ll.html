#include<iostream>
#include<cstdlib>
using namespace std;

struct Node
{
    int data;
    struct Node * next;
};

struct Node * createNode(int val){
    struct Node * ptr = (struct Node*)malloc(sizeof(struct Node));
    ptr->data = val;
    return ptr;


int main(){
    struct Node * head;    
    struct Node * second;    
    struct Node * third;    
    struct Node * forth;

    head = createNode(1);
    second = createNode(2);
    third = createNode(3);
    forth = createNode(4);

    head->next = second;

    second->next = third;

    third->next = forth;

    forth->next = NULL;

    return 0;
}

2. Insertion

//1. Insert at the begining
struct Node * insertStart(struct Node * head, int data){
    struct Node * p = createNode();
    p->data = data;
    p->next = head;
    return p;
}

//2. Insert at the end
struct Node * insertEnd(struct Node * head, int data){
    struct Node * ptr = createNode();
    ptr->data = data;
    struct Node * q = head;
    while (q->next != NULL)
    {
        q = q->next;
    }
    q->next = ptr;
    ptr->next = NULL;
    return head;
}

//3. Insert at particular index
struct Node * insertAt(struct Node * head, int posi, int data){
    struct Node * p = createNode();
    p->data = data;
    struct Node * q = head;
    int i = 0;
    while (i<posi-1)
    {
        q = q->next;
        i++;
    }
    p->next = q->next;
    q->next = p;
    return head;
}

//4. Insert after a node 
struct Node * insertAfter(struct Node * head, int datain, int newdata){
    struct Node * p = createNode();
    p->data = newdata;
    struct Node * q = head;
    while (q->data != datain)
    {
        q = q->next;
    }
    p->next = q->next;
    q->next = p;
    return head;
    
}

3. Deletion

//1. Delete the first node
struct Node * deleteStart(struct Node * head){
    struct Node * p = createNode();
    struct Node * q = head;
    p = q->next;
    free(q);
    return p;
}

//2.Delete the last node
struct Node * deleteEnd(struct Node * head){
    struct Node * p = head; 
    struct Node * q = head->next; 
    while (q->next != NULL)
    {
        p = p->next;
        q = q->next;
    }
    free(q);
    p->next = NULL;
    return head;
}

//3. Delete a node at particular index
struct Node * deleteAt(struct Node * head, int index){
    struct Node * p = head;
    struct Node * q  = head->next;
    int i = 0;
    while (i != (index-1))
    {
        p = p->next;
        q = q->next;
        i++;
    }
    p->next = q->next;
    free(q);
    return head;
}

//4. Delete a node after a particular node
struct Node * deleteAfter(struct Node * head, int value){
    struct Node * p = head;
    struct Node * q = head->next;
    while (p->data != value)
    {
        p = p->next;
        q = q->next;
    }
    p->next = q->next;
    free(q);
    return head;
}
